1 git add . // đẩy lên tất cả

2 git add * // đẩy lên tất cả

3 git commit -m "" // tiêu đề muốn pull lênh

4 git tag Name_tag muốn đẩy lên

5- git push origin master // đẩy lên nhánh master

6- git push origin Name_tag // đẩy lên tag muốn đưa lên

7- git init // tạo repostory trên git

8- git checkout -b feature_x  // tạo nhánh mới trên git

9 -git checkout master // trở lại nhánh 

10- git branch -d name_branch // xóa nhánh

11 -git branch // kiểm tra xem mình đang ở nhánh nào

12 -git branch -v // xem commit mới nhất trên nhánh

13- git merge tennhanh  // mới chuyển về nhánh master rồi mới merge

14 -git config --global user.name "YOUR NAME"

15-git config --global user.email "YOUR EMAIL ADDRESS"
